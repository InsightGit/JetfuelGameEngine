Detailed Description

template<class T>
class jetfuel::draw::Rect2d< T >

A simple 2d rectangle manipulation class.

Code Example: jetfuel::draw::Rect2d_int myrect(0,0,5,5); jetfuel::draw::Rectangle_2d_shape mydrawablerect(myrect); jetfuel::draw::Scene_manager scenemanager; jetfuel::draw::Scene scene1(1);

if(!scenemanager.Create_window("example", jetfuel::draw::Vector2d_int(0,0), jetfuel::draw::Vector2d_int(500,500))){ std::cerr << "[!]ERROR with creating sdl renderer!" << "Error is:" << SDL_GetError() << "\n"; }

if(!scenemanager.Create_renderer()){ std::cerr << "[!]ERROR with creating sdl renderer!" << "Error is:" << SDL_GetError() << "\n"; }

scenemanager.Switch_current_scene(&scene1);
